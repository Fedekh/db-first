Modellare la struttura di una tabella per memorizzare tutti i dati riguardanti delle auto usate messe in vendita da un concessionario.


id - SMALLINT - UNSIGNED - PRIMARY_KEY(UNIQUE - NOT NULL) - AUTO_INCREMENT
vin - VARCHAR(17) - UNIQUE - NOT NULL                (sarebbe il telaio/codice fiscale ed èunico)
car_brand - VARCHAR(15) - NOT NULL                   (Marchio auto)
car_model_name - VARCHAR(15) - NOT NULL              (Nome modello auto)
typing_car - VARCHAR(15)                             (Modello auto : se berlina, station wagon, suv...)
price - DECIMAL(6) - NOT NULL                        (ero del settore, e mai visto prezzi con decimali)
year - YEAR                                          (Anno della prima immatricolazione)
color -  VARCHAR(10)                                 (banalmente il colore)
metallic - TINYINT - DEFAULT(0)                      (se metallizzato o meno, default no)
fuel -  VARCHAR(10)                                  (tipo di carburante)
alloy wheels - TINYINT - DEFAULT(0)                  (cerchi in lega,default no)
spare_wheel - TINYINT - DEFAULT(0)                   (ruota di scorta,default no)
seats -   VARCHAR(8)                                 (posti)
doors -   VARCHAR(8)                                 (porte)
autoradio -   TINYINT - DEFAULT(0)                   (autoradio, scontato pero ,default no)
navigator -   TINYINT - DEFAULT(0)                   (navigatore,default no)
infotainment -TINYINT - DEFAULT(0)                   (applecarplay / android autoradio ,default no)
anti-puncture_kit - TINYINT - DEFAULT(0)             (kit antiforatura)
mileage - MEDIUMINT                                  (sempre per esperienza, difficilmente si vedranno auto usate in vendita oltre 300000 km)
owners - TINYINT (10)  - NOT NULL                    (numeri di proprietari che ha avuto)
imported - TINYINT - DEFAULT(0)                      (se è di importazione o meno, di default no)
available - TINYINT - DEFAULT(1)                     (se è disponibile, di default si)
vote_secure - TINYINT (5)                            (valutazione euro Ncap , da 1 a 5)
reviews - TEXT                                       (recensioni globali dei proprietari)

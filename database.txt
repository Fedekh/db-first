Modellare la struttura di una tabella per memorizzare tutti i dati riguardanti delle auto usate messe in vendita da un concessionario.


id - SMALLINT - UNSIGNED - PRIMARY_KEY(UNIQUE - NOT NULL) - AUTO_INCREMENT
vin - VARCHAR(17) - UNIQUE - NOT NULL                (sarebbe il telaio/codice fiscale ed èunico)
car_brand - VARCHAR(15) - NOT NULL                   (Marchio auto)
car_model_name - VARCHAR(15) - NOT NULL              (Nome modello auto)
typing_car - VARCHAR(15)                             (Modello auto : se berlina, station wagon, suv...)
price - MEDIUMINT - UNSIGNED - NOT NULL              (banalmente il prezzo)
year - YEAR                                          (Anno della prima immatricolazione)
color -  VARCHAR(10)                                 (banalmente il colore)
is_metallic - TINYINT - DEFAULT(0)                   (se metallizzato o meno, default no)
fuel -  VARCHAR(10)                                  (tipo di carburante)
cylinder -  TINYINT (12) - UNSIGNED                  (cilindrata, massimo V-12)
horse_power -   SMALLINT - UNSIGNED                  (cavalli potenza)
tax_horses - SMALLINT  - UNSIGNED                    (cavalli fiscali)
is_alloy_wheels - TINYINT - DEFAULT(0)               (cerchi in lega,default no)
is_spare_wheel - TINYINT - DEFAULT(0)                (ruota di scorta,default no)
seats - TINYINT (6) - UNSIGNED                       (posti)
doors -  TINYINT (6) - UNSIGNED                      (porte)
is_autoradio -   TINYINT - DEFAULT(0)                (autoradio, scontato pero ,default no)
is_navigator -   TINYINT - DEFAULT(0)                (navigatore,default no)
is_infotainment -TINYINT - DEFAULT(0)                (applecarplay / android autoradio ,default no)
anti-puncture_kit - TINYINT - DEFAULT(0)             (kit antiforatura)
mileage - MEDIUMINT  - UNSIGNED                      (sempre per esperienza, difficilmente si vedranno auto usate in vendita oltre 300000 km)
owners - TINYINT (10)  - NOT NULL                    (numeri di proprietari che ha avuto)
is_imported - TINYINT - DEFAULT(0)                   (se è di importazione o meno, di default no)
is_available - TINYINT - DEFAULT(1)                  (se è disponibile, di default si)
vote_secure - TINYINT (5) - UNSIGNED                 (valutazione euro Ncap , da 1 a 5)
reviews - TEXT                                       (recensioni globali dei proprietari)
